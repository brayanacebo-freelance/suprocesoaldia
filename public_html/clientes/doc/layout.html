Header

For the header component I think there will be no problem.

<!--header section-->
<header class="header">
    <!--nav bar helper-->
    <div class="navbar-helper">
        <!--we still recommend using a row-fluid to split the grid-->
        <div class="row-fluid">
            <!--grid site-name-->
            <div class="span2">
                <div class="panel-sitename">
                    <h2><a href="index.html"><span class="color-teal">Sti</span>learn</a></h2>
                </div>
            </div><!--/grid site-name-->

            <!--grid form search-->
            <div class="span6">
                <!--panel search-->
                <div class="panel-search">
                    <form class="form-search">
                        <div class="input-icon-append">
                            <button type="submit" rel="tooltip-bottom" title="search" class="icon">
                                <i class="icofont-search"></i>
                            </button>
                            <input class="input-large search-query grd-white" maxlength="23" placeholder="Search here..." type="text">
                        </div>
                    </form>
                </div><!--/panel search-->
            </div><!--/grid form search-->
            
            <!--grid button navigation-->
            <div class="span4">
                <!--panel button ext-->
                <div class="panel-ext">
                    <div class="btn-group">
                        <!--button-->
                        <a class="btn btn-inverse btn-small" href="#" title="lorem ipsum">Action</a>
                    </div>
                </div><!--/panel button ext-->
            </div><!--/grid button navigation-->

            <!--you don't have to split the display as seen above--> 
            <!--it would be more fun if adjusted to your taste-->
        </div> <!--/row-fluid--> 
    </div> <!--/nav bar helper--> 
</header> <!--/header-->

Section Content

I use grid system to divide the fluid column layout. Without changing any code in the bootstrap. Making it easy and simple to customize.

<!-- section content -->
<section class="section">
    <!--row-fluid-->
    <div class="row-fluid">
        <!-- span side-left -->
        <div class="span1">
        ...
        </div><!-- span side-left -->

        <!-- span content -->
        <div class="span9">
        ...
        </div><!-- /span content -->

        <!-- span side-right -->
        <div class="span2">
        ...
        </div><!-- /span side-right -->

    </div><!--/row-fluid -->
</section><!--/section content -->

If you want to use only two columns, just need to remove the right side. Don't forget to change the span content (span9) to (span11).

<!-- section content -->
<section class="section">
    <!--row-fluid-->
    <div class="row-fluid">
        <!-- span side-left -->
        <div class="span1">
        ...
        </div><!-- span side-left -->

        <!-- span content -->
        <div class="span11">
        ...
        </div><!-- /span content -->

    </div><!--/row-fluid -->
</section><!--/section content -->

Side Left

I use the side left to the menu, you just need to call class side-left <aside class="side-left">. It's a feature of stilearn, I have a calculation to determine the width, etc. It's the same calculation for class content and side-right. If you want to know how I did the calculations, please contact me via email.

Now you just add the template menu. Just like this :

<!-- span side-left -->
<div class="span1">
    <!--side bar-->
    <aside class="side-left">
        <!--side menu-->
        <ul class="sidebar">
            <!--always define class .first for first-child of li element sidebar left-->
            <li class="active first"> 
                <a href="#" title="dashboard">
                    <div class="helper-font-24">
                        <i class="icofont-dashboard"></i>
                    </div>
                    <span class="sidebar-text">Dashboard</span>
                </a>
            </li>
            <li>
                <a href="#" title="other menus">
                    <div class="helper-font-24">
                        <i class="icofont-th-large"></i>
                    </div>
                    <span class="sidebar-text">Other</span>
                </a>
            </li>
            ...
        </ul>
        <!--/side menu-->
    </aside><!--/side bar -->
</div><!-- span side-left -->

Content

Inside this content is contains the application that you created. Basically Format:

<!-- span content -->
<div class="span9">
    <!-- content -->
    <div class="content">
        <!-- content-header -->
        <div class="content-header">
            
            <h2><i class="icofont-home"></i> Dashboard <small>welcome to stilearn</small></h2>
            
        </div><!-- /content-header -->

        <!-- content-breadcrumb -->
        <div class="content-breadcrumb">
            
            <!--breadcrumb-->
            <ul class="breadcrumb">
                <li>
                    <a href="index.html">
                        <i class="icofont-home"></i> 
                        Dashboard
                    </a> 
                    <span class="divider">&rsaquo;</span>
                </li>
                <li class="active">Data elements</li>
            </ul><!--/breadcrumb-->
            
        </div><!-- /content-breadcrumb -->

        <!-- content-body -->
        <div class="content-body">

            //everything you want to create, write here
        
        </div><!--/content-body -->

    </div><!-- /content -->
</div><!-- /span content -->

Side Right

<div class="span2">
    <!-- side-right -->
    <aside class="side-right">
        <!-- sidebar-right -->
        <div class="sidebar-right">

                    ...

                </div><!-- /tab-content -->

        </div><!-- /sidebar-right -->
    </aside><!-- /side-right -->
</div><!-- /span side-right -->
